        POST /v1/verify

        // Express 中间件
async function requireKey(req, res, next) {
  const r = await fetch('https://keymgr.example.com/v1/verify', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ key: req.headers['x-api-key'] }),
  });
  // 200=ok, 其他状态码=失败（响应体里有 { valid:false, code, reason }）
  if (r.status !== 200) {
    const { code } = await r.json().catch(() => ({}));
    return res.status(r.status).json({ error: code || 'verify_failed' });
  }
  next();
}

        // net/http 中间件
func Verify(next http.Handler) http.Handler {
  return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
    key := r.Header.Get("X-API-Key")
    body, _ := json.Marshal(map[string]string{"key": key})
    resp, _ := http.Post("https://keymgr.example.com/v1/verify",
      "application/json", bytes.NewReader(body))
    defer resp.Body.Close()
    if resp.StatusCode != 200 {
      // 失败：响应体 { valid:false, code, reason }
      http.Error(w, "verify failed", resp.StatusCode)
      return
    }
    next.ServeHTTP(w, r)
  })
}

        行为说明

        • 鉴权：必须先通过 Key 验证（status 有效、未停用、未过期、且 key 不属于已软删账号），否则返回对应 code。
        • name 格式：2~64 字符 [A-Za-z0-9_.-]，建议大写+下划线（如 DB_URL、API_BASE）。
        • value 限制：字符串，≤ 8192 字符（足够存连接串、JSON 序列化配置等）。
        • 限流：1200 / 分钟/IP（/v1/verify 限流独立 600/min）。
        • 不记录 value：日志只记 name / key prefix，不记 value，避免日志泄露。
        • scope 访问控制：Key 的 meta.scopes 字段（数组）可限制该 Key 能读哪些变量。["*"] = 不限；["DB_URL","FEATURE_X"] = 仅能读这些；空数组 = 一个都读不了。被限制的变量在 /multi / /group 响应中出现在 forbidden 字段；/get 单独访问时返回 SCOPE_FORBIDDEN。
        • 主 Key 通杀：账号主 Key（isDefault=1）不受 scope 限制；admin 的主 Key 还可代任意账号身份调用（superKey=true）。

        状态码语义

        状态含义触发场景
        200OK成功
        400Bad Request参数缺失、JSON 解析失败、格式错；对外 API (/v1/*) 所有失败统一 400，错误类型用 code 字段区分
        401Unauthorized未登录或会话失效
        403Forbidden权限不足（非 admin 操作 admin 接口、跨账号操作别人的 key）
        404Not Found资源不存在（仅 /api/* 后台接口）
        409Conflictname 重复 / 哈希已存在 / 资源已被删除 / 最后一个 admin 保护
        422Unprocessable格式合法但语义错（key 太短、过期时间已过去）
        429Too Many Requests限流（verify 600/min/IP，var 1200/min/IP；登录 5/10min）
        503Service Unavailable健康检查：DB 不可用

        HTTPS 部署（生产环境必做）

        KeyMgr 本身只跑 HTTP（:3000），TLS 终止交给反代。反代会带上 X-Forwarded-Proto，驱动 COOKIE_SECURE=1 自动给 Cookie 加 Secure 标志。本机开发用 npm run dev 即可。

        # /etc/nginx/sites-enabled/keymgr.conf
# 申请证书：certbot --nginx -d keymgr.example.com
server {
  listen 443 ssl http2;
  server_name keymgr.example.com;

  ssl_certificate     /etc/letsencrypt/live/keymgr.example.com/fullchain.pem;
  ssl_certificate_key /etc/letsencrypt/live/keymgr.example.com/privkey.pem;

  # 收紧 TLS（推荐 Mozilla Intermediate）
  ssl_protocols TLSv1.2 TLSv1.3;
  ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-ECDSA-AES256-GCM-SHA384:ECDHE-RSA-AES256-GCM-SHA384;
  ssl_prefer_server_ciphers off;
  add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;

  client_max_body_size 2m;   # 限制上传，verify body 很小

  location / {
    proxy_pass         http://127.0.0.1:3000;
    proxy_http_version 1.1;
    proxy_set_header   Host              $host;
    proxy_set_header   X-Real-IP         $remote_addr;
    proxy_set_header   X-Forwarded-For   $proxy_add_x_forwarded_for;
    proxy_set_header   X-Forwarded-Proto $scheme;   # ← 关键：驱动 Secure Cookie
    proxy_set_header   X-Forwarded-Host  $host;
    proxy_read_timeout 30s;
  }
}

# 80 强制跳 443
server { listen 80; server_name keymgr.example.com; return 301 https://$host$request_uri; }
